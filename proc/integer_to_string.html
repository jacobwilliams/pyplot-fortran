    subroutine integer_to_string(i, s)

    integer,          intent(in), optional  :: i     !! integer value

    integer :: istat !! IO status

    write(s, int_fmt, iostat=istat) i

    if (istat/=0) then
        write(error_unit,'(A)') 'Error converting integer to string'
        s = '****'
    else
        s = adjustl(s)
    end if
